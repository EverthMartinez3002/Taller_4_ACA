    Configuración del Entorno de Desarrollo#

    Esta guía te ayudará a configurar un entorno de desarrollo completo para trabajar con CupcakeAPI, incluyendo herramientas de desarrollo, testing y deployment.

    🚀 Configuración Inicial#

    Requisitos del Sistema#

    Obligatorios#

    • Node.js 18.0+ (para herramientas de desarrollo)
    • Git 2.30+ (control de versiones)
    • Editor de código (VS Code recomendado)
    • Terminal/CLI con soporte para comandos modernos

    Recomendados#

    • Docker 20.0+ (contenedores para desarrollo)
    • Postman o Insomnia (testing de APIs)
    • ngrok (túneles para webhooks)

    Variables de Entorno#

    Crea un archivo .env en la raíz de tu proyecto:

    # CupcakeAPI Configuración
CUPCAKE_API_BASE_URL=https://api.cupcakeapi.com/v1
CUPCAKE_API_KEY=your_api_key_here
CUPCAKE_API_SECRET=your_api_secret_here

# Entorno de desarrollo
NODE_ENV=development
DEBUG=cupcakeapi:*

# Base de datos (si desarrollas localmente)
DATABASE_URL=postgresql://user:password@localhost:5432/cupcakeapi_dev
REDIS_URL=redis://localhost:6379

# Servicios externos
STRIPE_PUBLISHABLE_KEY=pk_test_...
STRIPE_SECRET_KEY=sk_test_...
TWILIO_ACCOUNT_SID=AC...
TWILIO_AUTH_TOKEN=...
SENDGRID_API_KEY=SG...

# Webhooks y túneles
WEBHOOK_SECRET=your_webhook_secret
NGROK_AUTH_TOKEN=your_ngrok_token

# Testing
TEST_API_KEY=your_test_api_key
TEST_USER_EMAIL=test@cupcakeapi.com

    Cliente API Base#

    Crea un cliente base para CupcakeAPI:

    src/api/client.js 

    class CupcakeAPIClient {
    constructor(config = {}) {
        this.baseURL = config.baseURL || process.env.CUPCAKE_API_BASE_URL;
        this.apiKey = config.apiKey || process.env.CUPCAKE_API_KEY;
        this.userToken = null;
        this.retryConfig = {
            maxRetries: 3,
            retryDelay: 1000,
            retryOn: [408, 429, 500, 502, 503, 504]
        };
    }

    setUserToken(token) {
        this.userToken = token;
    }

    async request(endpoint, options = {}) {
        const url = `${this.baseURL}${endpoint}`;
        const config = {
            headers: {
                'Authorization': `Bearer ${this.userToken || this.apiKey}`,
                'Content-Type': 'application/json',
                'User-Agent': 'CupcakeApp/1.0.0',
                ...options.headers
            },
            ...options
        };

        return this.requestWithRetry(url, config);
    }

    async requestWithRetry(url, config, retryCount = 0) {
        try {
            const response = await fetch(url, config);

            if (!response.ok) {
                const errorData = await response.json().catch(() => ({}));
                throw new CupcakeAPIError(response, errorData);
            }

            return await response.json();
        } catch (error) {
            if (this.shouldRetry(error, retryCount)) {
                await this.delay(this.retryConfig.retryDelay * Math.pow(2, retryCount));
                return this.requestWithRetry(url, config, retryCount + 1);
            }
            throw error;
        }
    }

    shouldRetry(error, retryCount) {
        return retryCount < this.retryConfig.maxRetries &&
               this.retryConfig.retryOn.includes(error.status);
    }

    delay(ms) {
        return new Promise(resolve => setTimeout(resolve, ms));
    }

    // Métodos de conveniencia
    async get(endpoint, params = {}) {
        const query = new URLSearchParams(params).toString();
        const url = query ? `${endpoint}?${query}` : endpoint;
        return this.request(url);
    }

    async post(endpoint, data = {}) {
        return this.request(endpoint, {
            method: 'POST',
            body: JSON.stringify(data)
        });
    }

    async put(endpoint, data = {}) {
        return this.request(endpoint, {
            method: 'PUT',
            body: JSON.stringify(data)
        });
    }

    async delete(endpoint) {
        return this.request(endpoint, { method: 'DELETE' });
    }
}

class CupcakeAPIError extends Error {
    constructor(response, data) {
        super(data.error?.message || `HTTP ${response.status}`);
        this.name = 'CupcakeAPIError';
        this.status = response.status;
        this.code = data.error?.code;
        this.details = data.error?.details;
        this.requestId = data.request_id;
    }
}

// Instancia global
const cupcakeAPI = new CupcakeAPIClient();

export { CupcakeAPIClient, CupcakeAPIError, cupcakeAPI };

    Módulos por Endpoint#

    src/api/endpoints/cupcakes.js 

    import { cupcakeAPI } from '../client.js';

export const cupcakesAPI = {
    async getAll(filters = {}) {
        return cupcakeAPI.get('/cupcakes', filters);
    },

    async getById(id) {
        return cupcakeAPI.get(`/cupcakes/${id}`);
    },

    async search(query, filters = {}) {
        return cupcakeAPI.get('/cupcakes/search', { q: query, ...filters });
    },

    async getPopular(period = 'week') {
        return cupcakeAPI.get('/cupcakes/popular', { period });
    },

    async getCategories() {
        return cupcakeAPI.get('/cupcakes/categories');
    },

    // Métodos admin (requieren permisos)
    async create(cupcakeData) {
        return cupcakeAPI.post('/cupcakes', cupcakeData);
    },

    async update(id, cupcakeData) {
        return cupcakeAPI.put(`/cupcakes/${id}`, cupcakeData);
    },

    async updateStock(id, stockData) {
        return cupcakeAPI.patch(`/cupcakes/${id}/stock`, stockData);
    },

    async delete(id) {
        return cupcakeAPI.delete(`/cupcakes/${id}`);
    }
};

    Hook de React para Estado#

    src/hooks/useCupcakeAPI.js 

    import { useState, useEffect, useCallback } from 'react';
import { cupcakesAPI } from '../api/endpoints/cupcakes.js';

export function useCupcakes(filters = {}) {
    const [cupcakes, setCupcakes] = useState([]);
    const [loading, setLoading] = useState(true);
    const [error, setError] = useState(null);
    const [pagination, setPagination] = useState(null);

    const fetchCupcakes = useCallback(async () => {
        try {
            setLoading(true);
            setError(null);

            const response = await cupcakesAPI.getAll(filters);

            setCupcakes(response.data.cupcakes);
            setPagination(response.pagination);
        } catch (err) {
            setError(err);
        } finally {
            setLoading(false);
        }
    }, [filters]);

    useEffect(() => {
        fetchCupcakes();
    }, [fetchCupcakes]);

    const refresh = useCallback(() => {
        fetchCupcakes();
    }, [fetchCupcakes]);

    return {
        cupcakes,
        loading,
        error,
        pagination,
        refresh
    };
}

export function useCupcake(id) {
    const [cupcake, setCupcake] = useState(null);
    const [loading, setLoading] = useState(true);
    const [error, setError] = useState(null);

    useEffect(() => {
        if (!id) return;

        const fetchCupcake = async () => {
            try {
                setLoading(true);
                setError(null);

                const response = await cupcakesAPI.getById(id);
                setCupcake(response.data.cupcake);
            } catch (err) {
                setError(err);
            } finally {
                setLoading(false);
            }
        };

        fetchCupcake();
    }, [id]);

    return { cupcake, loading, error };
}

    Test de Ejemplo#

    tests/api/cupcakes.test.js 

    import { cupcakesAPI } from '../../src/api/endpoints/cupcakes.js';
import { testCupcakeAPI, testHelpers } from '../setup.js';

describe('Cupcakes API', () => {
    describe('getAll', () => {
        test('should fetch cupcakes successfully', async () => {
            const response = await cupcakesAPI.getAll();

            expect(response.status).toBe('success');
            expect(response.data.cupcakes).toBeInstanceOf(Array);
            expect(response.pagination).toBeDefined();
        });

        test('should apply filters correctly', async () => {
            const filters = { category: 'premium', limit: 5 };
            const response = await cupcakesAPI.getAll(filters);

            expect(response.data.cupcakes).toHaveLength(5);
            response.data.cupcakes.forEach(cupcake => {
                expect(cupcake.category).toBe('premium');
            });
        });
    });

    describe('search', () => {
        test('should return relevant results', async () => {
            const response = await cupcakesAPI.search('chocolate');

            expect(response.data.cupcakes).toBeInstanceOf(Array);
            response.data.cupcakes.forEach(cupcake => {
                expect(
                    cupcake.name.toLowerCase().includes('chocolate') ||
                    cupcake.description.toLowerCase().includes('chocolate')
                ).toBe(true);
            });
        });
    });
});

describe('Cupcakes API Integration', () => {
    let testCupcakeId;

    beforeAll(async () => {
        // Crear cupcake de prueba si tienes permisos
        try {
            const testCupcake = testHelpers.createTestCupcake();
            const response = await testCupcakeAPI.post('/cupcakes', testCupcake);
            testCupcakeId = response.data.cupcake.id;
        } catch (error) {
            console.log('No se pudo crear cupcake de prueba:', error.message);
        }
    });

    afterAll(async () => {
        // Limpiar cupcake de prueba
        if (testCupcakeId) {
            try {
                await testCupcakeAPI.delete(`/cupcakes/${testCupcakeId}`);
            } catch (error) {
                console.log('No se pudo eliminar cupcake de prueba:', error.message);
            }
        }
    });

    test('should handle API rate limiting gracefully', async () => {
        // Test de rate limiting
        const promises = Array(10).fill().map(() => cupcakesAPI.getAll());

        const results = await Promise.allSettled(promises);

        // Al menos algunas deberían ser exitosas
        const successful = results.filter(r => r.status === 'fulfilled');
        expect(successful.length).toBeGreaterThan(0);
    });
});

    Logger Personalizado#

    src/utils/logger.js 

    class Logger {
    constructor(namespace = 'app') {
        this.namespace = namespace;
        this.enabled = process.env.DEBUG?.includes(namespace) || process.env.NODE_ENV === 'development';
    }

    log(level, message, data = {}) {
        if (!this.enabled) return;

        const timestamp = new Date().toISOString();
        const prefix = `[${timestamp}] [${this.namespace}] [${level.toUpperCase()}]`;

        console.log(prefix, message);

        if (Object.keys(data).length > 0) {
            console.log(JSON.stringify(data, null, 2));
        }
    }

    info(message, data) {
        this.log('info', message, data);
    }

    error(message, data) {
        this.log('error', message, data);
    }

    debug(message, data) {
        this.log('debug', message, data);
    }

    warn(message, data) {
        this.log('warn', message, data);
    }

    apiRequest(method, url, data = {}) {
        this.debug(`API Request: ${method} ${url}`, data);
    }

    apiResponse(url, response, duration) {
        this.debug(`API Response: ${url} (${duration}ms)`, {
            status: response.status,
            data: response.data
        });
    }
}

export const logger = new Logger('cupcakeapi');
export const createLogger = (namespace) => new Logger(namespace);

    Interceptor de Requests#

    src/utils/apiInterceptor.js 

    import { logger } from './logger.js';

export function createAPIInterceptor(client) {
    const originalRequest = client.request.bind(client);

    client.request = async function(endpoint, options = {}) {
        const startTime = Date.now();
        const method = options.method || 'GET';

        logger.apiRequest(method, endpoint, {
            headers: options.headers,
            body: options.body
        });

        try {
            const response = await originalRequest(endpoint, options);
            const duration = Date.now() - startTime;

            logger.apiResponse(endpoint, response, duration);

            return response;
        } catch (error) {
            const duration = Date.now() - startTime;

            logger.error(`API Error: ${method} ${endpoint} (${duration}ms)`, {
                error: error.message,
                status: error.status,
                code: error.code
            });

            throw error;
        }
    };

    return client;
}
