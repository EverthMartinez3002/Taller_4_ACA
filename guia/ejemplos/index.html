    🌐 Aplicación Web de E-commerce#

    Caso de Uso: Tienda Online de Cupcakes#

    Una aplicación web completa donde los usuarios pueden navegar, personalizar y ordenar cupcakes.

    Flujo Principal#

    class CupcakeStore {
    constructor(apiKey) {
        this.apiKey = apiKey;
        this.baseURL = 'https://api.cupcakeapi.com/v1';
        this.userToken = localStorage.getItem('user_token');
    }

    // 1. Cargar catálogo de cupcakes
    async loadCatalog(filters = {}) {
        const params = new URLSearchParams(filters);
        const response = await fetch(`${this.baseURL}/cupcakes?${params}`, {
            headers: { 'Authorization': `Bearer ${this.apiKey}` }
        });

        const data = await response.json();
        return data.data.cupcakes;
    }

    // 2. Mostrar detalles de un cupcake
    async getCupcakeDetails(cupcakeId) {
        const response = await fetch(`${this.baseURL}/cupcakes/${cupcakeId}`, {
            headers: { 'Authorization': `Bearer ${this.apiKey}` }
        });

        return await response.json();
    }

    // 3. Agregar al carrito (localStorage)
    addToCart(cupcake, quantity, customizations = {}) {
        let cart = JSON.parse(localStorage.getItem('cart') || '[]');

        const item = {
            id: cupcake.id,
            name: cupcake.name,
            price: cupcake.price,
            quantity,
            customizations,
            subtotal: cupcake.price * quantity
        };

        cart.push(item);
        localStorage.setItem('cart', JSON.stringify(cart));
        this.updateCartUI();
    }

    // 4. Procesar pedido
    async checkout(orderData) {
        if (!this.userToken) {
            throw new Error('Usuario debe estar autenticado');
        }

        const response = await fetch(`${this.baseURL}/orders`, {
            method: 'POST',
            headers: {
                'Authorization': `Bearer ${this.userToken}`,
                'Content-Type': 'application/json'
            },
            body: JSON.stringify(orderData)
        });

        if (!response.ok) {
            throw new Error('Error al procesar el pedido');
        }

        const result = await response.json();

        // Limpiar carrito después del pedido exitoso
        localStorage.removeItem('cart');

        return result.data.order;
    }
}

// Uso práctico
const store = new CupcakeStore('tu_api_key');

// Cargar productos en la página principal
async function initHomePage() {
    try {
        const cupcakes = await store.loadCatalog({ 
            category: 'premium', 
            limit: 8,
            sort: 'popularity' 
        });

        renderCupcakeGrid(cupcakes);
    } catch (error) {
        showErrorMessage('Error al cargar productos');
    }
}

// Manejar compra
async function handlePurchase() {
    const cart = JSON.parse(localStorage.getItem('cart') || '[]');

    const orderData = {
        items: cart.map(item => ({
            cupcake_id: item.id,
            quantity: item.quantity,
            customizations: item.customizations
        })),
        delivery: {
            type: 'delivery',
            address: getUserAddress(),
            scheduled_date: getSelectedDeliveryTime()
        },
        payment: {
            method: 'credit_card',
            card_token: await getPaymentToken()
        }
    };

    try {
        const order = await store.checkout(orderData);
        showSuccessMessage(`Pedido ${order.id} creado exitosamente`);
        redirectToOrderConfirmation(order.id);
    } catch (error) {
        showErrorMessage('Error al procesar el pedido');
    }
}

    Caso de Uso: App de Pedidos para Delivery#

    import React, { useState, useEffect } from 'react';
import { View, Text, ScrollView, TouchableOpacity, Image, Alert } from 'react-native';
import AsyncStorage from '@react-native-async-storage/async-storage';

const CupcakeApp = () => {
    const [cupcakes, setCupcakes] = useState([]);
    const [cart, setCart] = useState([]);
    const [userToken, setUserToken] = useState(null);

    const API_KEY = 'tu_api_key';
    const BASE_URL = 'https://api.cupcakeapi.com/v1';

    useEffect(() => {
        loadUserToken();
        loadCupcakes();
        loadCart();
    }, []);

    const loadUserToken = async () => {
        const token = await AsyncStorage.getItem('user_token');
        setUserToken(token);
    };

    const loadCupcakes = async () => {
        try {
            const response = await fetch(`${BASE_URL}/cupcakes?available=true&limit=20`, {
                headers: { 'Authorization': `Bearer ${API_KEY}` }
            });

            const data = await response.json();
            setCupcakes(data.data.cupcakes);
        } catch (error) {
            Alert.alert('Error', 'No se pudieron cargar los cupcakes');
        }
    };

    const addToCart = async (cupcake, quantity = 1) => {
        const newItem = {
            id: cupcake.id,
            name: cupcake.name,
            price: cupcake.price,
            quantity,
            image: cupcake.images[0]
        };

        const updatedCart = [...cart, newItem];
        setCart(updatedCart);

        await AsyncStorage.setItem('cart', JSON.stringify(updatedCart));
        Alert.alert('Éxito', 'Producto agregado al carrito');
    };

    const checkout = async () => {
        if (!userToken) {
            Alert.alert('Error', 'Debes iniciar sesión para realizar un pedido');
            return;
        }

        const orderData = {
            items: cart.map(item => ({
                cupcake_id: item.id,
                quantity: item.quantity
            })),
            delivery: {
                type: 'delivery',
                address: await getUserAddress()
            }
        };

        try {
            const response = await fetch(`${BASE_URL}/orders`, {
                method: 'POST',
                headers: {
                    'Authorization': `Bearer ${userToken}`,
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify(orderData)
            });

            const result = await response.json();

            if (response.ok) {
                Alert.alert('¡Pedido Exitoso!', `Tu pedido ${result.data.order.id} ha sido creado`);
                setCart([]);
                await AsyncStorage.removeItem('cart');
            } else {
                Alert.alert('Error', result.error.message);
            }
        } catch (error) {
            Alert.alert('Error', 'No se pudo procesar el pedido');
        }
    };

    return (
        <ScrollView style={{ flex: 1, padding: 16 }}>
            <Text style={{ fontSize: 24, fontWeight: 'bold', marginBottom: 16 }}>
                Sweet Dreams Cupcakes
            </Text>

            {cupcakes.map(cupcake => (
                <View key={cupcake.id} style={{
                    backgroundColor: 'white',
                    borderRadius: 8,
                    padding: 16,
                    marginBottom: 16,
                    shadowColor: '#000',
                    shadowOffset: { width: 0, height: 2 },
                    shadowOpacity: 0.1,
                    shadowRadius: 4
                }}>
                    <Image 
                        source={{ uri: cupcake.images[0] }}
                        style={{ width: '100%', height: 200, borderRadius: 8 }}
                    />
                    <Text style={{ fontSize: 18, fontWeight: 'bold', marginTop: 8 }}>
                        {cupcake.name}
                    </Text>
                    <Text style={{ color: '#666', marginVertical: 4 }}>
                        {cupcake.description}
                    </Text>
                    <Text style={{ fontSize: 20, fontWeight: 'bold', color: '#ff6b9d' }}>
                        ${cupcake.price.toFixed(2)}
                    </Text>
                    <TouchableOpacity
                        style={{
                            backgroundColor: '#ff6b9d',
                            padding: 12,
                            borderRadius: 6,
                            marginTop: 8
                        }}
                        onPress={() => addToCart(cupcake)}
                    >
                        <Text style={{ color: 'white', textAlign: 'center', fontWeight: 'bold' }}>
                            Agregar al Carrito
                        </Text>
                    </TouchableOpacity>
                </View>
            ))}

            {cart.length > 0 && (
                <TouchableOpacity
                    style={{
                        backgroundColor: '#28a745',
                        padding: 16,
                        borderRadius: 6,
                        marginTop: 16
                    }}
                    onPress={checkout}
                >
                    <Text style={{ color: 'white', textAlign: 'center', fontSize: 18, fontWeight: 'bold' }}>
                        Proceder al Checkout ({cart.length} items)
                    </Text>
                </TouchableOpacity>
            )}
        </ScrollView>
    );
};

export default CupcakeApp;

    Caso de Uso: Dashboard Administrativo#

    import requests
import streamlit as st
import pandas as pd
import plotly.express as px
from datetime import datetime, timedelta

class CupcakeDashboard:
    def __init__(self, api_key, manager_token):
        self.api_key = api_key
        self.manager_token = manager_token
        self.base_url = 'https://api.cupcakeapi.com/v1'

    def get_headers(self, use_manager_token=False):
        token = self.manager_token if use_manager_token else self.api_key
        return {
            'Authorization': f'Bearer {token}',
            'Content-Type': 'application/json'
        }

    def get_dashboard_data(self, period='month'):
        """Obtiene datos del dashboard"""
        response = requests.get(
            f'{self.base_url}/analytics/dashboard?period={period}',
            headers=self.get_headers(use_manager_token=True)
        )
        return response.json()['data']

    def get_orders(self, status=None, limit=50):
        """Obtiene lista de pedidos"""
        params = {'limit': limit}
        if status:
            params['status'] = status

        response = requests.get(
            f'{self.base_url}/orders',
            params=params,
            headers=self.get_headers(use_manager_token=True)
        )
        return response.json()['data']['orders']

    def update_order_status(self, order_id, new_status):
        """Actualiza el estado de un pedido"""
        response = requests.patch(
            f'{self.base_url}/orders/{order_id}/status',
            json={'status': new_status},
            headers=self.get_headers(use_manager_token=True)
        )
        return response.status_code == 200

    def get_inventory(self):
        """Obtiene estado del inventario"""
        response = requests.get(
            f'{self.base_url}/inventory',
            headers=self.get_headers(use_manager_token=True)
        )
        return response.json()['data']['items']

    Caso de Uso: Bot para Consultas y Pedidos#

    import openai
import requests
from flask import Flask, request, jsonify

app = Flask(__name__)

class CupcakeChatbot:
    def __init__(self, api_key, openai_key):
        self.api_key = api_key
        self.base_url = 'https://api.cupcakeapi.com/v1'
        openai.api_key = openai_key

    def get_cupcakes_info(self):
        """Obtiene información de cupcakes para el contexto del bot"""
        response = requests.get(
            f'{self.base_url}/cupcakes?limit=20',
            headers={'Authorization': f'Bearer {self.api_key}'}
        )
        return response.json()['data']['cupcakes']

    def search_cupcakes(self, query):
        """Busca cupcakes específicos"""
        response = requests.get(
            f'{self.base_url}/cupcakes',
            params={'search': query, 'limit': 5},
            headers={'Authorization': f'Bearer {self.api_key}'}
        )
        return response.json()['data']['cupcakes']

    def get_order_status(self, order_id, user_token):
        """Obtiene el estado de un pedido"""
        response = requests.get(
            f'{self.base_url}/orders/{order_id}',
            headers={'Authorization': f'Bearer {user_token}'}
        )
        if response.status_code == 200:
            return response.json()['data']['order']
        return None

    def generate_response(self, user_message, context=None):
        """Genera respuesta usando OpenAI"""
        cupcakes = self.get_cupcakes_info()

        system_prompt = f"""
        Eres un asistente virtual para Sweet Dreams Cupcakes. 
        Ayudas a los clientes con información sobre productos, pedidos y consultas generales.

        Información de productos disponibles:
        {[f"{c['name']}: {c['description']} - ${c['price']}" for c in cupcakes[:10]]}

        Responde de manera amigable y útil. Si el cliente pregunta por un producto específico,
        proporciona detalles como precio, ingredientes y opciones de personalización.
        """

        response = openai.ChatCompletion.create(
            model="gpt-3.5-turbo",
            messages=[
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": user_message}
            ]
        )

        return response.choices[0].message.content

bot = CupcakeChatbot(
    api_key="tu_api_key",
    openai_key="tu_openai_key"
)

@app.route('/chat', methods=['POST'])
def chat():
    data = request.json
    user_message = data.get('message')
    user_token = data.get('user_token')  # Opcional para consultas de pedidos

    # Detectar intención del mensaje
    if 'pedido' in user_message.lower() and 'estado' in user_message.lower():
        # Extraer ID del pedido del mensaje
        # Lógica simplificada - en producción usar NLP más robusto
        words = user_message.split()
        order_id = None
        for word in words:
            if word.startswith('ORD-'):
                order_id = word
                break

        if order_id and user_token:
            order = bot.get_order_status(order_id, user_token)
            if order:
                return jsonify({
                    'response': f"Tu pedido {order_id} está en estado: {order['status']}. "
                              f"Total: ${order['pricing']['total']:.2f}"
                })

    # Respuesta general usando IA
    response = bot.generate_response(user_message)

    return jsonify({'response': response})

if __name__ == '__main__':
    app.run(debug=True)

        function addMessage(message, isUser = false) {
            const messageDiv = document.createElement('div');
            messageDiv.className = `message ${isUser ? 'user-message' : 'bot-message'}`;
            messageDiv.textContent = message;
            messagesContainer.appendChild(messageDiv);
            messagesContainer.scrollTop = messagesContainer.scrollHeight;
        }

        async function sendMessage() {
            const message = messageInput.value.trim();
            if (!message) return;

            addMessage(message, true);
            messageInput.value = '';

            try {
                const response = await fetch('/chat', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ 
                        message,
                        user_token: localStorage.getItem('user_token') 
                    })
                });

                const data = await response.json();
                addMessage(data.response);
            } catch (error) {
                addMessage('Lo siento, ocurrió un error. Intenta nuevamente.');
            }
        }

        messageInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') {
                sendMessage();
            }
        });

        // Mensaje de bienvenida
        addMessage('¡Hola! Soy tu asistente virtual. Puedo ayudarte con información sobre nuestros cupcakes, el estado de tu pedido, y responder tus consultas. ¿En qué puedo ayudarte?');
    </script>

    // Google Apps Script para automatizar reportes
function updateDailySalesReport() {
  const API_KEY = 'tu_api_key';
  const MANAGER_TOKEN = 'tu_manager_token';
  const BASE_URL = 'https://api.cupcakeapi.com/v1';

  // Obtener datos del día
  const today = new Date();
  const startDate = Utilities.formatDate(today, 'GMT', 'yyyy-MM-dd');

  const response = UrlFetchApp.fetch(`${BASE_URL}/analytics/sales?start_date=${startDate}&end_date=${startDate}`, {
    headers: {
      'Authorization': `Bearer ${MANAGER_TOKEN}`
    }
  });

  const salesData = JSON.parse(response.getContentText());

  // Obtener la hoja de cálculo
  const sheet = SpreadsheetApp.getActiveSheet();

  // Agregar nueva fila con datos del día
  sheet.appendRow([
    startDate,
    salesData.data.total_revenue,
    salesData.data.total_orders,
    salesData.data.average_order_value,
    salesData.data.total_cupcakes_sold
  ]);

  // Enviar email con resumen
  sendDailySummaryEmail(salesData.data);
}

function sendDailySummaryEmail(data) {
  const subject = `Resumen de Ventas - ${new Date().toLocaleDateString()}`;
  const body = `
    Resumen de ventas del día:

    • Ingresos totales: $${data.total_revenue.toFixed(2)}
    • Pedidos procesados: ${data.total_orders}
    • Valor promedio por pedido: $${data.average_order_value.toFixed(2)}
    • Cupcakes vendidos: ${data.total_cupcakes_sold}

    ¡Que tengas un buen día!
  `;

  MailApp.sendEmail('gerente@cupcakestore.com', subject, body);
}

// Configurar trigger para ejecutar diariamente
function createDailyTrigger() {
  ScriptApp.newTrigger('updateDailySalesReport')
    .timeBased()
    .everyDays(1)
    .atHour(9) // 9 AM
    .create();
}
