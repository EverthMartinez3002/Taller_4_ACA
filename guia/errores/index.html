    Gestión de Errores#

    Esta guía te ayudará a entender, manejar y resolver errores de manera efectiva al integrar CupcakeAPI en tu aplicación.

    Estructura de Respuestas de Error#

    CupcakeAPI utiliza códigos de estado HTTP estándar y proporciona información detallada en el cuerpo de la respuesta para ayudarte a diagnosticar y resolver problemas.

    Formato de Error Estándar#

    {
    "status": "error",
    "error": {
        "code": "ERROR_CODE",
        "message": "Descripción legible del error",
        "details": "Información adicional para resolver el problema",
        "field_errors": {
            "campo": ["Error específico del campo"]
        },
        "documentation_url": "https://docs.cupcakeapi.com/errors/ERROR_CODE"
    },
    "request_id": "req_1234567890abcdef",
    "timestamp": "2024-01-20T10:15:30Z"
}

    📋 Códigos de Estado HTTP#

    2xx - Éxito#

    Código Descripción Uso
    200 OK Petición exitosa con datos
    201 Created Recurso creado exitosamente
    204 No Content Petición exitosa sin datos de respuesta

    4xx - Errores del Cliente#

    Código Descripción Causa Común Solución
    400 Bad Request Parámetros inválidos Verificar formato de datos
    401 Unauthorized Token inválido/faltante Renovar o verificar autenticación
    403 Forbidden Permisos insuficientes Verificar rol del usuario
    404 Not Found Recurso no existe Verificar ID del recurso
    422 Unprocessable Entity Errores de validación Revisar campos requeridos
    429 Too Many Requests Rate limit excedido Implementar backoff

    5xx - Errores del Servidor#

    Código Descripción Acción Recomendada
    500 Internal Server Error Reintentar después de un tiempo
    502 Bad Gateway Verificar estado del servicio
    503 Service Unavailable Implementar retry con backoff

    🔍 Códigos de Error Específicos#

    Autenticación (AUTH_*)#

    AUTH_INVALID_TOKEN#

    {
    "status": "error",
    "error": {
        "code": "AUTH_INVALID_TOKEN",
        "message": "El token proporcionado no es válido",
        "details": "El token JWT ha sido modificado o es inválido"
    }
}

    Causas: - Token malformado - Token modificado - Clave secreta incorrecta

    Solución: 

    // Verificar formato del token
if (!token || !token.includes('.')) {
    throw new Error('Token malformado');
}

// Renovar token
const newToken = await refreshAuthToken();

    AUTH_TOKEN_EXPIRED#

    {
    "error": {
        "code": "AUTH_TOKEN_EXPIRED",
        "message": "El token JWT ha expirado",
        "details": "Utiliza el refresh_token para obtener uno nuevo"
    }
}

    Solución Automática: 

    async function makeAPIRequest(url, options = {}) {
    let response = await fetch(url, options);

    if (response.status === 401) {
        const errorData = await response.json();

        if (errorData.error.code === 'AUTH_TOKEN_EXPIRED') {
            // Renovar token automáticamente
            const newToken = await refreshToken();

            // Reintentar petición original
            options.headers.Authorization = `Bearer ${newToken}`;
            response = await fetch(url, options);
        }
    }

    return response;
}

    Implementar Backoff Exponencial: 

    class APIClient {
    constructor() {
        this.retryDelays = [1000, 2000, 4000, 8000]; // ms
    }

    async makeRequest(url, options = {}, retryCount = 0) {
        try {
            const response = await fetch(url, options);

            if (response.status === 429) {
                if (retryCount < this.retryDelays.length) {
                    const delay = this.retryDelays[retryCount];
                    console.log(`Rate limit reached. Retrying in ${delay}ms...`);

                    await new Promise(resolve => setTimeout(resolve, delay));
                    return this.makeRequest(url, options, retryCount + 1);
                } else {
                    throw new Error('Max retries exceeded');
                }
            }

            return response;
        } catch (error) {
            throw error;
        }
    }
}

    🛠️ Implementación de Manejo de Errores#

    Clase de Error Personalizada#

    class CupcakeAPIError extends Error {
    constructor(response, data) {
        super(data.error.message);
        this.name = 'CupcakeAPIError';
        this.code = data.error.code;
        this.status = response.status;
        this.details = data.error.details;
        this.fieldErrors = data.error.field_errors;
        this.requestId = data.request_id;
    }

    isAuthError() {
        return this.code.startsWith('AUTH_');
    }

    isValidationError() {
        return this.code.startsWith('VALIDATION_');
    }

    isRateLimitError() {
        return this.code === 'RATE_LIMIT_EXCEEDED';
    }
}

    Cliente API con Manejo de Errores#

    class CupcakeAPIClient {
    constructor(apiKey) {
        this.apiKey = apiKey;
        this.baseURL = 'https://api.cupcakeapi.com/v1';
        this.userToken = null;
    }

    async request(endpoint, options = {}) {
        const url = `${this.baseURL}${endpoint}`;
        const config = {
            headers: {
                'Authorization': `Bearer ${this.userToken || this.apiKey}`,
                'Content-Type': 'application/json',
                ...options.headers
            },
            ...options
        };

        try {
            const response = await fetch(url, config);
            const data = await response.json();

            if (!response.ok) {
                throw new CupcakeAPIError(response, data);
            }

            return data;
        } catch (error) {
            if (error instanceof CupcakeAPIError) {
                this.handleAPIError(error);
                throw error;
            }

            // Error de red u otro tipo
            console.error('Network or unexpected error:', error);
            throw new Error('Error de conexión. Verifica tu internet.');
        }
    }

    handleAPIError(error) {
        // Log para debugging
        console.error(`API Error [${error.code}]:`, {
            message: error.message,
            status: error.status,
            requestId: error.requestId
        });

        // Manejo específico por tipo de error
        if (error.isAuthError()) {
            this.handleAuthError(error);
        } else if (error.isValidationError()) {
            this.handleValidationError(error);
        } else if (error.isRateLimitError()) {
            this.handleRateLimitError(error);
        }
    }

    handleAuthError(error) {
        switch (error.code) {
            case 'AUTH_TOKEN_EXPIRED':
                this.refreshToken();
                break;
            case 'AUTH_INSUFFICIENT_PERMISSIONS':
                this.redirectToUpgrade();
                break;
            default:
                this.redirectToLogin();
        }
    }

    handleValidationError(error) {
        if (error.fieldErrors) {
            this.displayFieldErrors(error.fieldErrors);
        }
    }

    handleRateLimitError(error) {
        // Implementar retry automático o mostrar mensaje
        this.showRateLimitMessage();
    }
}

    Sistema de Notificaciones de Error#

    class ErrorNotificationSystem {
    constructor() {
        this.createErrorContainer();
    }

    createErrorContainer() {
        const container = document.createElement('div');
        container.id = 'error-notifications';
        container.style.cssText = `
            position: fixed;
            top: 20px;
            right: 20px;
            z-index: 10000;
            max-width: 400px;
        `;
        document.body.appendChild(container);
    }

    showError(error, type = 'error') {
        const notification = document.createElement('div');
        notification.className = `notification notification-${type}`;
        notification.style.cssText = `
            background: ${type === 'error' ? '#f44336' : '#ff9800'};
            color: white;
            padding: 16px;
            margin-bottom: 10px;
            border-radius: 4px;
            box-shadow: 0 2px 8px rgba(0,0,0,0.2);
            animation: slideInRight 0.3s ease;
        `;

        notification.innerHTML = `
            <div style="display: flex; justify-content: space-between; align-items: start;">
                <div>
                    <strong>${this.getErrorTitle(error)}</strong>
                    <p style="margin: 4px 0 0 0; font-size: 14px;">${error.message}</p>
                    ${error.details ? `<small style="opacity: 0.8;">${error.details}</small>` : ''}
                </div>
                <button onclick="this.parentElement.parentElement.remove()" style="background: none; border: none; color: white; font-size: 18px; cursor: pointer;">×</button>
            </div>
        `;

        document.getElementById('error-notifications').appendChild(notification);

        // Auto-remove después de 5 segundos
        setTimeout(() => {
            if (notification.parentElement) {
                notification.remove();
            }
        }, 5000);
    }

    getErrorTitle(error) {
        const titles = {
            'AUTH_INVALID_TOKEN': '🔐 Error de Autenticación',
            'AUTH_TOKEN_EXPIRED': '⏰ Sesión Expirada',
            'VALIDATION_REQUIRED_FIELD': '📝 Campos Requeridos',
            'RESOURCE_NOT_FOUND': '🔍 No Encontrado',
            'RATE_LIMIT_EXCEEDED': '⚡ Demasiadas Peticiones',
            'RESOURCE_OUT_OF_STOCK': '📦 Sin Stock'
        };

        return titles[error.code] || '❌ Error';
    }
}

// Uso global
const errorNotifications = new ErrorNotificationSystem();

// Manejo global de errores no capturados
window.addEventListener('unhandledrejection', (event) => {
    if (event.reason instanceof CupcakeAPIError) {
        errorNotifications.showError(event.reason);
        event.preventDefault(); // Prevenir que aparezca en consola
    }
});

    // Mock para simular errores durante desarrollo
class MockCupcakeAPI {
    constructor(errorRate = 0.1) {
        this.errorRate = errorRate;
        this.errors = [
            { status: 401, code: 'AUTH_TOKEN_EXPIRED' },
            { status: 429, code: 'RATE_LIMIT_EXCEEDED' },
            { status: 422, code: 'VALIDATION_REQUIRED_FIELD' }
        ];
    }

    async request(endpoint, options) {
        // Simular latencia de red
        await new Promise(resolve => setTimeout(resolve, Math.random() * 500));

        // Simular errores aleatoriamente
        if (Math.random() < this.errorRate) {
            const error = this.errors[Math.floor(Math.random() * this.errors.length)];
            throw new CupcakeAPIError({ status: error.status }, {
                error: { code: error.code, message: 'Error simulado' }
            });
        }

        // Respuesta exitosa simulada
        return { status: 'success', data: {} };
    }
}

    📊 Monitoreo y Logging#

    Sistema de Logs#

    class ErrorLogger {
    constructor() {
        this.logs = [];
        this.maxLogs = 1000;
    }

    log(error, context = {}) {
        const logEntry = {
            timestamp: new Date().toISOString(),
            error: {
                code: error.code,
                message: error.message,
                status: error.status,
                requestId: error.requestId
            },
            context,
            userAgent: navigator.userAgent,
            url: window.location.href
        };

        this.logs.unshift(logEntry);

        // Mantener solo los últimos N logs
        if (this.logs.length > this.maxLogs) {
            this.logs = this.logs.slice(0, this.maxLogs);
        }

        // Enviar a servicio de análisis
        this.sendToAnalytics(logEntry);
    }

    sendToAnalytics(logEntry) {
        // Integración con Google Analytics, Sentry, etc.
        if (window.gtag) {
            gtag('event', 'api_error', {
                error_code: logEntry.error.code,
                error_status: logEntry.error.status
            });
        }
    }

    getErrorStats() {
        const stats = {};
        this.logs.forEach(log => {
            const code = log.error.code;
            stats[code] = (stats[code] || 0) + 1;
        });
        return stats;
    }
}

const errorLogger = new ErrorLogger();

    🚀 Mejores Prácticas#

    1. Retry Strategy#

    async function retryWithBackoff(fn, maxRetries = 3) {
    for (let i = 0; i < maxRetries; i++) {
        try {
            return await fn();
        } catch (error) {
            if (i === maxRetries - 1 || !isRetryableError(error)) {
                throw error;
            }

            const delay = Math.pow(2, i) * 1000; // Exponential backoff
            await new Promise(resolve => setTimeout(resolve, delay));
        }
    }
}

function isRetryableError(error) {
    return error.status >= 500 || error.code === 'RATE_LIMIT_EXCEEDED';
}

    2. Circuit Breaker Pattern#

    class CircuitBreaker {
    constructor(threshold = 5, timeout = 60000) {
        this.failureThreshold = threshold;
        this.timeout = timeout;
        this.failureCount = 0;
        this.lastFailureTime = null;
        this.state = 'CLOSED'; // CLOSED, OPEN, HALF_OPEN
    }

    async call(fn) {
        if (this.state === 'OPEN') {
            if (Date.now() - this.lastFailureTime > this.timeout) {
                this.state = 'HALF_OPEN';
            } else {
                throw new Error('Circuit breaker is OPEN');
            }
        }

        try {
            const result = await fn();
            this.onSuccess();
            return result;
        } catch (error) {
            this.onFailure();
            throw error;
        }
    }

    onSuccess() {
        this.failureCount = 0;
        this.state = 'CLOSED';
    }

    onFailure() {
        this.failureCount++;
        this.lastFailureTime = Date.now();

        if (this.failureCount >= this.failureThreshold) {
            this.state = 'OPEN';
        }
    }
}

    3. Error Boundaries en React#

    class ErrorBoundary extends React.Component {
    constructor(props) {
        super(props);
        this.state = { hasError: false, error: null };
    }

    static getDerivedStateFromError(error) {
        return { hasError: true, error };
    }

    componentDidCatch(error, errorInfo) {
        if (error instanceof CupcakeAPIError) {
            errorLogger.log(error, { component: this.props.component });
        }
    }

    render() {
        if (this.state.hasError) {
            return (
                <div className="error-fallback">
                    <h2>🧁 Oops! Algo salió mal</h2>
                    <p>No pudimos cargar este contenido. Por favor intenta nuevamente.</p>
                    <button onClick={() => this.setState({ hasError: false })}>
                        Reintentar
                    </button>
                </div>
            );
        }

        return this.props.children;
    }
}
