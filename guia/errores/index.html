    Implementar Backoff Exponencial: 

    class APIClient {
    constructor() {
        this.retryDelays = [1000, 2000, 4000, 8000]; // ms
    }

    async makeRequest(url, options = {}, retryCount = 0) {
        try {
            const response = await fetch(url, options);

            if (response.status === 429) {
                if (retryCount < this.retryDelays.length) {
                    const delay = this.retryDelays[retryCount];
                    console.log(`Rate limit reached. Retrying in ${delay}ms...`);

                    await new Promise(resolve => setTimeout(resolve, delay));
                    return this.makeRequest(url, options, retryCount + 1);
                } else {
                    throw new Error('Max retries exceeded');
                }
            }

            return response;
        } catch (error) {
            throw error;
        }
    }
}

    Clase de Error Personalizada#

    class CupcakeAPIError extends Error {
    constructor(response, data) {
        super(data.error.message);
        this.name = 'CupcakeAPIError';
        this.code = data.error.code;
        this.status = response.status;
        this.details = data.error.details;
        this.fieldErrors = data.error.field_errors;
        this.requestId = data.request_id;
    }

    isAuthError() {
        return this.code.startsWith('AUTH_');
    }

    isValidationError() {
        return this.code.startsWith('VALIDATION_');
    }

    isRateLimitError() {
        return this.code === 'RATE_LIMIT_EXCEEDED';
    }
}

    Cliente API con Manejo de Errores#

    class CupcakeAPIClient {
    constructor(apiKey) {
        this.apiKey = apiKey;
        this.baseURL = 'https://api.cupcakeapi.com/v1';
        this.userToken = null;
    }

    async request(endpoint, options = {}) {
        const url = `${this.baseURL}${endpoint}`;
        const config = {
            headers: {
                'Authorization': `Bearer ${this.userToken || this.apiKey}`,
                'Content-Type': 'application/json',
                ...options.headers
            },
            ...options
        };

        try {
            const response = await fetch(url, config);
            const data = await response.json();

            if (!response.ok) {
                throw new CupcakeAPIError(response, data);
            }

            return data;
        } catch (error) {
            if (error instanceof CupcakeAPIError) {
                this.handleAPIError(error);
                throw error;
            }

            // Error de red u otro tipo
            console.error('Network or unexpected error:', error);
            throw new Error('Error de conexi√≥n. Verifica tu internet.');
        }
    }

    handleAPIError(error) {
        // Log para debugging
        console.error(`API Error [${error.code}]:`, {
            message: error.message,
            status: error.status,
            requestId: error.requestId
        });

        // Manejo espec√≠fico por tipo de error
        if (error.isAuthError()) {
            this.handleAuthError(error);
        } else if (error.isValidationError()) {
            this.handleValidationError(error);
        } else if (error.isRateLimitError()) {
            this.handleRateLimitError(error);
        }
    }

    handleAuthError(error) {
        switch (error.code) {
            case 'AUTH_TOKEN_EXPIRED':
                this.refreshToken();
                break;
            case 'AUTH_INSUFFICIENT_PERMISSIONS':
                this.redirectToUpgrade();
                break;
            default:
                this.redirectToLogin();
        }
    }

    handleValidationError(error) {
        if (error.fieldErrors) {
            this.displayFieldErrors(error.fieldErrors);
        }
    }

    handleRateLimitError(error) {
        // Implementar retry autom√°tico o mostrar mensaje
        this.showRateLimitMessage();
    }
}

    Sistema de Logs#

    class ErrorLogger {
    constructor() {
        this.logs = [];
        this.maxLogs = 1000;
    }

    log(error, context = {}) {
        const logEntry = {
            timestamp: new Date().toISOString(),
            error: {
                code: error.code,
                message: error.message,
                status: error.status,
                requestId: error.requestId
            },
            context,
            userAgent: navigator.userAgent,
            url: window.location.href
        };

        this.logs.unshift(logEntry);

        // Mantener solo los √∫ltimos N logs
        if (this.logs.length > this.maxLogs) {
            this.logs = this.logs.slice(0, this.maxLogs);
        }

        // Enviar a servicio de an√°lisis
        this.sendToAnalytics(logEntry);
    }

    sendToAnalytics(logEntry) {
        // Integraci√≥n con Google Analytics, Sentry, etc.
        if (window.gtag) {
            gtag('event', 'api_error', {
                error_code: logEntry.error.code,
                error_status: logEntry.error.status
            });
        }
    }

    getErrorStats() {
        const stats = {};
        this.logs.forEach(log => {
            const code = log.error.code;
            stats[code] = (stats[code] || 0) + 1;
        });
        return stats;
    }
}

    1. Retry Strategy#

    async function retryWithBackoff(fn, maxRetries = 3) {
    for (let i = 0; i < maxRetries; i++) {
        try {
            return await fn();
        } catch (error) {
            if (i === maxRetries - 1 || !isRetryableError(error)) {
                throw error;
            }

            const delay = Math.pow(2, i) * 1000; // Exponential backoff
            await new Promise(resolve => setTimeout(resolve, delay));
        }
    }
}

function isRetryableError(error) {
    return error.status >= 500 || error.code === 'RATE_LIMIT_EXCEEDED';
}

    2. Circuit Breaker Pattern#

    class CircuitBreaker {
    constructor(threshold = 5, timeout = 60000) {
        this.failureThreshold = threshold;
        this.timeout = timeout;
        this.failureCount = 0;
        this.lastFailureTime = null;
        this.state = 'CLOSED'; // CLOSED, OPEN, HALF_OPEN
    }

    async call(fn) {
        if (this.state === 'OPEN') {
            if (Date.now() - this.lastFailureTime > this.timeout) {
                this.state = 'HALF_OPEN';
            } else {
                throw new Error('Circuit breaker is OPEN');
            }
        }

        try {
            const result = await fn();
            this.onSuccess();
            return result;
        } catch (error) {
            this.onFailure();
            throw error;
        }
    }

    onSuccess() {
        this.failureCount = 0;
        this.state = 'CLOSED';
    }

    onFailure() {
        this.failureCount++;
        this.lastFailureTime = Date.now();

        if (this.failureCount >= this.failureThreshold) {
            this.state = 'OPEN';
        }
    }
}

    3. Error Boundaries en React#

    class ErrorBoundary extends React.Component {
    constructor(props) {
        super(props);
        this.state = { hasError: false, error: null };
    }

    static getDerivedStateFromError(error) {
        return { hasError: true, error };
    }

    componentDidCatch(error, errorInfo) {
        if (error instanceof CupcakeAPIError) {
            errorLogger.log(error, { component: this.props.component });
        }
    }

    render() {
        if (this.state.hasError) {
            return (
                <div className="error-fallback">
                    <h2>üßÅ Oops! Algo sali√≥ mal</h2>
                    <p>No pudimos cargar este contenido. Por favor intenta nuevamente.</p>
                    <button onClick={() => this.setState({ hasError: false })}>
                        Reintentar
                    </button>
                </div>
            );
        }

        return this.props.children;
    }
}
